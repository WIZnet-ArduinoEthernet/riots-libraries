276 /* Converts Arduino pin to right AVR PORT */
277 #define __pinToPort(P) \
278 (((P) <= 7) ? &PORTD : (((P) <= 13) ? &PORTB : &PORTC))
279 
280 /* Converts Arduino pin to right AVR DDR */

284 /* Converts Arduino pin to right AVR PIN */

288 /* Converts Arduino pin to right pin bit */
289 #define __pinToBit(P) \
290 (((P) <= 7) ? (P) : (((P) <= 13) ? (P) - 8 : (P) - 14))
